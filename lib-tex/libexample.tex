\section{基本输入输出标准库\code{stdio}}
\subsection{\code{fgets()}}
\begin{minted}{c}
/* [name:fgets.c] */
#include <stdlib.h>
#include <unistd.h>
#include <stdio.h>
int main()
{
	char buf[200];
	FILE *fp = NULL;
	if (access("./fgets.c", R_OK)) {
		printf("No such file!\n");
		exit(-1);
	}

	fp = fopen("./fgets.c", "r");
	while(fgets(buf, 200, fp))
		printf("%s", buf);
	fclose(fp);

	printf("Please input one line here: ");
	fgets(buf, 200, stdin);
	printf("%s", buf);

	return 0;
}
\end{minted}
